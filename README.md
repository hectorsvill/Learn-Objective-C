###  swiftch case

*Swift*
```swift
func learn2SwitchCase() {
	let n = 42
	switch n {
	case 1...19:
		print("a number between 1...19")
	case 20...42:
		let x = 10
		print("found 42 and x = \(x)")
	default:
		print("default")
	}
}

```

*Objective-C*
```objective-c

void learn2SwitchCase() {
	int n = 42;
	switch (n) {
		case 1:
		{
			int foo = 1;
			NSLog(@"found %i", foo);
			break;
		}
		case 42:
			NSLog(@"Found %d \n", n);
	}
}
```
